window.removerMembro = async (id, nome) => { const overlay = document.createElement("div"); overlay.className = "logout-overlay"; const caixa = document.createElement("div"); caixa.className = "logout-caixa"; caixa.innerHTML = `

`; overlay.appendChild(caixa); document.body.appendChild(overlay); document.getElementById("cancelarRemocao").onclick = () => overlay.remove(); document.getElementById("confirmarRemocao").onclick = async () => { await db.collection("senhas").doc(id).delete(); overlay.remove(); alert(`Membro ${nome} removido com sucesso!`); carregarMembros(); }; }; };
